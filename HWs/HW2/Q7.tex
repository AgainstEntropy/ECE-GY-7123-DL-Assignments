\noindent \textred{7.}
Given 
\[ 
    \min(x, y) = \left\{
    \begin{aligned}
        &y &, x > y \\
        &x &, y > x
    \end{aligned}
    \right.
\]
Find $\frac{d}{dy} \frac{d}{dx} \min(x, y)$. \\
\myAnswer{
\begin{align*}
    \min(x, y) &= y \cdot u(x-y) + x \cdot u(y-x) \\
    \Longrightarrow \frac{d}{dy} \frac{d}{dx} \min(x, y) &= \frac{d}{dx} [y \delta(x-y) + u(y-x) - x \cdot \delta(y-x)] \\
    ~ &= \frac{d}{dy} u(y-x) = \underline{\delta(y-x)}
\end{align*}
}
