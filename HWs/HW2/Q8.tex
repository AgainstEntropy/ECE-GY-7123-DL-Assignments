\noindent \textred{8.}
A coin with $P(h) = p = 1-q$ is tossed $n$ times. Find the probability that the number of heads is even.
\myAnswer{
\begin{align*}
    1 = 1^n = (p + q)^n &= \sum_{k=0}^n C_n^k p^k q^{n-k} \\
    &= \sum_{k_1=0}^{\lfloor{n/2}\rfloor} C_n^{2k_1} p^{2k_1} q^{n-2k_1} + \sum_{k_2=0}^{\lceil{n/2}\rceil} C_n^{2k_2 + 1} p^{2k_2 + 1} q^{n-2k_2 + 1} \\
    &= P(k=\text{even}) + P(k=\text{odd})
\end{align*}
\begin{align*}
    (q - p)^n &= \sum_{k=0}^n C_n^k (-p)^k q^{n-k} \\
    &= \sum_{k_1=0}^{\lfloor{n/2}\rfloor} C_n^{2k_1} (-p)^{2k_1} q^{n-2k_1} + \sum_{k_2=0}^{\lceil{n/2}\rceil} C_n^{2k_2 + 1} (-p)^{2k_2 + 1} q^{n-2k_2 + 1} \\
    &= P(k=\text{even}) - P(k=\text{odd})
\end{align*}
Therefore, we can readily know that $P(k=\text{even}) = \underline{[1 + (q-p)^n]/2}$.
}
