    \item[(2)] $E\{X\} = \int_{-\infty}^{\infty} x f(x) dx$. Find $E\{X\}$. \\
    \myAnswer{
    \begin{align*}
        E\{X\} &= 0.25 \left\{ \int_{-\infty}^{\infty} x \delta(x+2) dx + \int_{-\infty}^{\infty} x \delta(x+1) dx + \int_{-\infty}^{\infty} x [u(x) - u(x-2)] dx] \right\} \\
        &= 0.25 \times [ (-2) + (-1) + 2 ] = \underline{-0.25}
    \end{align*}
    }
